package "Model (Entidades — SQLite)" as Model #E8F8F5 {

  class users <<entity>> {
    + id: TEXT (PK)
    + name: TEXT
    + email: TEXT (UNIQUE)
    + phone: TEXT (UNIQUE)
    + avatar_url: TEXT
    + google_id: TEXT (UNIQUE)
    + role: TEXT (client|provider|admin)
    + banned: INTEGER
    + deleted_at: TEXT
  }

  class provider_profiles <<entity>> {
    + id: TEXT (PK)
    + user_id: TEXT (FK→users, UNIQUE)
    + category_id: TEXT (FK→categories)
    + description: TEXT
    + rating: REAL
    + review_count: INTEGER
    + latitude: REAL
    + longitude: REAL
    + city: TEXT
    + state: TEXT
    + active: INTEGER
    + experience_years: INTEGER
    + service_radius_km: REAL
    + address: TEXT
    + contact_count: INTEGER
  }

  class categories <<entity>> {
    + id: TEXT (PK)
    + name: TEXT
    + slug: TEXT (UNIQUE)
    + icon: TEXT
    + deleted_at: TEXT
  }

  class provider_categories <<entity>> {
    + provider_id: TEXT (PK,FK)
    + category_id: TEXT (PK,FK)
  }

  class service_requests <<entity>> {
    + id: TEXT (PK)
    + client_id: TEXT (FK→users)
    + title: TEXT
    + description: TEXT
    + category_id: TEXT (FK→categories)
    + city: TEXT
    + state: TEXT
    + latitude: REAL
    + longitude: REAL
    + status: TEXT
    + urgency: TEXT
    + max_value: REAL
    + photo_url: TEXT
    + address: TEXT
  }

  class interests <<entity>> {
    + id: TEXT (PK)
    + request_id: TEXT (FK→service_requests)
    + provider_id: TEXT (FK→users)
  }

  class reviews <<entity>> {
    + id: TEXT (PK)
    + client_id: TEXT (FK→users)
    + provider_id: TEXT (FK→provider_profiles)
    + contact_id: TEXT (FK→contact_history)
    + rating: INTEGER (1-5)
    + comment: TEXT
    + response: TEXT
  }

  class portfolio_photos <<entity>> {
    + id: TEXT (PK)
    + provider_id: TEXT (FK→provider_profiles)
    + filename: TEXT
    + original_name: TEXT
    + mime_type: TEXT
    + size_bytes: INTEGER
    + tag: TEXT (Antes|Depois|Geral)
    + sort_order: INTEGER
  }

  class provider_wizard_state <<entity>> {
    + id: TEXT (PK)
    + user_id: TEXT (FK→users, UNIQUE)
    + current_step: INTEGER
    + step_data: TEXT (JSON)
  }

  class contact_history <<entity>> {
    + id: TEXT (PK)
    + client_id: TEXT (FK→users)
    + provider_id: TEXT (FK→users)
    + contact_type: TEXT
  }

  class favorites <<entity>> {
    + id: TEXT (PK)
    + client_id: TEXT (FK→users)
    + provider_id: TEXT (FK→users)
  }

  class notifications <<entity>> {
    + id: TEXT (PK)
    + user_id: TEXT (FK→users)
    + type: TEXT
    + title: TEXT
    + body: TEXT
    + data: TEXT (JSON)
    + read: INTEGER
  }

  class notification_preferences <<entity>> {
    + user_id: TEXT (PK,FK→users)
    + new_requests: INTEGER
    + interests: INTEGER
    + reviews: INTEGER
    + promotions: INTEGER
  }

  class reports <<entity>> {
    + id: TEXT (PK)
    + reporter_id: TEXT (FK→users)
    + target_type: TEXT
    + target_id: TEXT
    + reason: TEXT
    + description: TEXT
    + status: TEXT
    + resolved_by: TEXT (FK→users)
    + resolution: TEXT
  }

  class bans <<entity>> {
    + id: TEXT (PK)
    + user_id: TEXT (FK→users)
    + admin_id: TEXT (FK→users)
    + reason: TEXT
    + revoked: INTEGER
    + revoked_by: TEXT
  }

  class admin_actions <<entity>> {
    + id: TEXT (PK)
    + admin_id: TEXT (FK→users)
    + action_type: TEXT
    + target_type: TEXT
    + target_id: TEXT
    + justification: TEXT
  }

  class refresh_tokens <<entity>> {
    + id: TEXT (PK)
    + user_id: TEXT (FK→users)
    + token: TEXT (UNIQUE)
    + expires_at: TEXT
    + revoked: INTEGER
  }

  class otp_codes <<entity>> {
    + id: TEXT (PK)
    + phone: TEXT
    + code: TEXT
    + expires_at: TEXT
    + used: INTEGER
    + identifier_type: TEXT
  }

  class blacklisted_tokens <<entity>> {
    + id: TEXT (PK)
    + token_jti: TEXT (UNIQUE)
    + expires_at: TEXT
  }

  class recovery_tokens <<entity>> {
    + id: TEXT (PK)
    + user_id: TEXT (FK→users)
    + identifier: TEXT
    + token: TEXT (UNIQUE)
    + expires_at: TEXT
    + used: INTEGER
  }

package "Controller (Rotas Express)" as Controller #E8EAF6 {

  class AuthController <<auth.ts>> {
    + POST /google — Login Google OAuth
    + GET /google/url — URL de autenticação
    + GET /google/callback — Callback OAuth
    + POST /otp/send — Enviar código OTP
    + POST /otp/verify — Verificar OTP
    + POST /refresh — Renovar JWT
    + POST /logout — Revogar tokens
    + GET /me — Dados do usuário logado
    + POST /recover/send — Recuperação passo 1
    + POST /recover/verify — Recuperação passo 2
    + POST /recover/reset — Recuperação passo 3
    + DELETE /account — Excluir conta
  }

  class SearchController <<search.ts>> {
    + GET /categories — Listar categorias
    + GET /providers/search — Buscar prestadores
  }

  class RequestsController <<requests.ts>> {
    + POST / — Criar pedido
    + GET / — Listar pedidos (admin)
    + GET /mine — Meus pedidos
    + PATCH /:id — Editar/cancelar pedido
    + POST /:id/interest — Demonstrar interesse
    + GET /:id/interests — Ver interesses
    + GET /open — Mural de pedidos
  }

  class ProviderController <<provider.ts>> {
    + GET /wizard — Estado do wizard
    + PUT /wizard — Salvar passo do wizard
    + POST /wizard/complete — Finalizar cadastro
    + GET /me — Meu perfil prestador
    + POST /portfolio/upload — Upload foto
    + GET /:id/portfolio — Portfólio público
    + DELETE /portfolio/:photoId — Remover foto
    + GET /me/portfolio — Meu portfólio
  }

  class ProvidersController <<providers.ts>> {
    + GET /me — Status do prestador
    + PATCH /me/status — Alternar disponibilidade
    + GET /profile/mine — Perfil completo
    + GET /:id — Perfil público
    + PUT /profile — Editar perfil
    + POST /portfolio — Upload portfólio
    + DELETE /portfolio/:id — Remover foto
    + POST /:userId/reviews — Avaliar
    + GET /:userId/reviews — Listar avaliações
    + POST /:userId/report — Denunciar
  }

  class NotificationsController <<notifications.ts>> {
    + GET / — Listar notificações
    + PATCH /:id/read — Marcar lida
    + PATCH /read-all — Marcar todas lidas
    + GET /preferences — Ver preferências
    + PUT /preferences — Salvar preferências
  }

  class FavoritesController <<favorites.ts>> {
    + GET / — Meus favoritos
    + POST /:providerId — Toggle favorito
    + GET /check/:providerId — Verificar se é favorito
  }

  class ContactsController <<contacts.ts>> {
    + GET / — Histórico de contatos
    + POST / — Registrar novo contato
  }

  class AdminController <<admin.ts>> {
    + GET /categories — Listar categorias
    + POST /categories — Criar categoria
    + PUT /categories/:id — Editar categoria
    + DELETE /categories/:id — Remover categoria
    + GET /stats — Dashboard métricas
    + GET /reports — Listar denúncias
    + GET /reports/:id — Detalhe denúncia
    + POST /reports/:id/resolve — Resolver denúncia
    + POST /users/:id/ban — Banir usuário
    + POST /users/:id/unban — Revogar ban
    + GET /bans — Listar bans ativos
  }
}
