package "Model — Entidades SQLite" #E8F8F5 {

  class users {
    id | name | email | phone | avatar_url | google_id
    role: client | provider | admin
    banned | deleted_at
  }

  class provider_profiles {
    id | user_id(FK) | category_id(FK)
    description | rating | review_count
    city | state | lat | lng
    active | experience_years | service_radius_km | address
  }

  class categories {
    id | name | slug | icon
  }

  class provider_categories <<M:N>> {
    provider_id | category_id
  }

  class service_requests {
    id | client_id(FK) | category_id
    title | description | status
    city | state | urgency | max_value
  }

  class interests {
    id | request_id(FK) | provider_id(FK)
  }

  class reviews {
    id | client_id | provider_id
    rating(1-5) | comment | response
  }

  class portfolio_photos {
    id | provider_id(FK)
    filename | tag: Antes|Depois|Geral
  }

  class contact_history {
    id | client_id | provider_id
  }

  class favorites {
    id | client_id | provider_id
  }

  class notifications {
    id | user_id | type | title | read
  }

  class notification_preferences {
    user_id | new_requests | interests | reviews | promotions
  }

  class reports {
    id | reporter_id | reason | status
  }

  class bans {
    id | user_id | admin_id | reason | revoked
  }

  ' Relacionamentos

  users ||--o{ favorites

  provider_profiles }o--|| categories

  provider_categories }o--|| categories
  provider_profiles ||--o{ portfolio_photos
  provider_profiles ||--o{ reviews

  service_requests ||--o{ interests
  interests }o--|| users
}

package "DAO — Acesso a Dados (db.ts)" #FCF3CF {

  class CategoryDAO {
    getAll() | create() | update() | softDelete()
    getWithProviderCount()
  }

  class ProviderDAO {
    search(filters) | count(filters)
    getProfile(userId) | createProfile(userId, data)
    setCategories() | getCategories()
  }

  class PortfolioDAO {
    addPhoto() | getPhotos() | getPhoto()
    deletePhoto() | countPhotos()
  }

  class WizardDAO {
    getState() | createState()
    updateState() | deleteState()
  }

  class RequestDAO {
    searchOpen(filters)
  }

  class ReviewDAO {
    getForProvider() | createReview()
  }

  class ReportDAO {
    createReport() | hasRecentReport()
    getPending()
  }

  class AdminDAO {
    getStats(periodDays)
  }

  CategoryDAO ..> categories
  ProviderDAO ..> provider_profiles
  PortfolioDAO ..> portfolio_photos
  WizardDAO ..> provider_profiles
  RequestDAO ..> service_requests
  ReviewDAO ..> reviews
  ReportDAO ..> reports
  AdminDAO ..> users
}

' ═══════════════════════ CONTROLLER ═══════════════════════

package "Controller — Rotas Express" #E8EAF6 {

  class "auth.ts" as AuthCtl {
    POST /google | /otp/send | /otp/verify
    POST /refresh | /logout | /recover/*
    GET /me | DELETE /account
  }

  class "search.ts" as SearchCtl {
    GET /categories
    GET /providers/search
  }

  class "requests.ts" as ReqCtl {
    POST / (criar) | GET /mine
    PATCH /:id (editar/cancelar)
    GET /open (mural)
    POST /:id/interest | GET /:id/interests
  }

  class "provider.ts" as ProvCtl {
    GET/PUT /wizard | POST /wizard/complete
    POST /portfolio/upload | DELETE /portfolio/:id
  }

  class "providers.ts" as ProvsCtl {
    GET /:id (perfil público)
    PUT /profile | PATCH /me/status
    POST /:id/reviews | POST /:id/report
  }

  class "notifications.ts" as NotifCtl {
    GET / | PATCH /:id/read | PATCH /read-all
    GET/PUT /preferences
  }

  class "favorites.ts\ncontacts.ts" as MiscCtl {
    Favorites: GET / | POST /:id (toggle) | GET /check/:id
    Contacts: GET / | POST /
  }

  class "admin.ts" as AdminCtl {
    CRUD /categories | GET /stats
    GET/POST /reports | POST /:id/resolve
    POST /users/:id/ban | /unban
  }
}
