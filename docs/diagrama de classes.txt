package "Model (SQLite)" #E8F8F5 {

  class users {
    id, name, email, phone
    avatar_url, google_id
    role (client|provider|admin)
    banned, deleted_at
  }

  class provider_profiles {
    id, user_id (FK)
    category_id (FK)
    description, rating, review_count
    city, state, latitude, longitude
    active, experience_years
    service_radius_km, address
  }

  class categories {
    id, name, slug, icon
  }

  class provider_categories <<M:N>> {
    provider_id, category_id
  }

  class service_requests {
    id, client_id (FK)
    title, description, category_id
    city, state, status
    urgency, max_value, photo_url
  }

  class interests {
    id, request_id (FK)
    provider_id (FK)
  }

  class reviews {
    id, client_id, provider_id
    rating (1-5), comment, response
  }

  class portfolio_photos {
    id, provider_id (FK)
    filename, tag (Antes|Depois|Geral)
  }

  class favorites {
    id, client_id, provider_id
  }

  class contact_history {
    id, client_id, provider_id
  }

  class notifications {
    id, user_id, type, title
    read
  }

  class notification_preferences {
    user_id
    new_requests, interests
    reviews, promotions
  }

  class reports {
    id, reporter_id, reason
    status (pending|resolved)
  }

  class bans {
    id, user_id, admin_id, reason
    revoked
  }

  ' ── Relacionamentos ──
  users ||--o| provider_profiles : "1"
  users ||--o{ service_requests : "publica"
  users ||--o{ reviews : "avalia"
  users ||--o{ favorites : "favorita"
  users ||--o{ contact_history : "contata"
  users ||--o{ notifications : "recebe"
  users ||--|| notification_preferences : "configura"
  users ||--o{ reports : "denuncia"
  users ||--o{ bans : "banido"

  provider_profiles }o--|| categories : "categoria"
  provider_profiles ||--o{ provider_categories : ""
  provider_categories }o--|| categories : ""
  provider_profiles ||--o{ portfolio_photos : "portfólio"
  provider_profiles ||--o{ reviews : "recebe"

  service_requests ||--o{ interests : "interesses"
  interests }o--|| users : "prestador"
}

package "Controller (Express)" #E8EAF6 {

  class "AuthController" as AuthCtl {
    POST /google, /otp/send, /otp/verify
    POST /refresh, /logout
    POST /recover/send → verify → reset
    GET /me, DELETE /account
  }

  class "SearchController" as SearchCtl {
    GET /categories
    GET /providers/search
  }

  class "RequestsController" as ReqCtl {
    POST / (criar), GET /mine
    PATCH /:id (editar/cancelar)
    GET /open (mural)
    POST /:id/interest, GET /:id/interests
  }

  class "ProviderController" as ProvCtl {
    GET/PUT /wizard, POST /wizard/complete
    POST /portfolio/upload, DELETE /portfolio/:id
  }

  class "ProvidersController" as ProvsCtl {
    GET /:id (perfil público)
    PUT /profile (editar)
    POST /:id/reviews, POST /:id/report
  }

  class "NotificationsController" as NotifCtl {
    GET /, PATCH /:id/read, PATCH /read-all
    GET/PUT /preferences
  }

  class "Favorites + Contacts" as MiscCtl {
    Favorites: GET /, POST /:id (toggle)
    Contacts: GET /, POST /
  }

  class "AdminController" as AdminCtl {
    CRUD /categories
    GET /stats, GET /reports, POST /reports/:id/resolve
    POST /users/:id/ban, POST /users/:id/unban
  }
}
